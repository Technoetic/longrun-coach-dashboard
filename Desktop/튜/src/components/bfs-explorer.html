의사코드

frontier = Queue()
frontier.put(start)
came_from = {start: None}
 
while not frontier.empty():
  current = frontier.get()
  if current == goal: break
  for next in neighbors(current):
    if next not in came_from:
      frontier.put(next)
      came_from[next] = current
